@startuml
start
:User opens app;
:Browse products;
if (Found desired product?) then (yes)
  :Add to cart;
  :Review cart;
  :Proceed to checkout;
else (no)
  :Continue browsing;
  :Browse products;
endif
:Enter shipping address;
:Select shipping method;
:Enter payment info;
if (Payment valid?) then (yes)
  :Process payment;
  :Confirm order;
  :Send confirmation email;
else (no)
  :Show error;
  :Return to payment;
endif
:Order placed;
stop
@enduml
